Three ways to connect JS to your page.

Linking JavaScript

<!-- 1. External file (recommended) — place before </body> --> <script src="app.js" defer></script> <!-- 2. Inline script — for small snippets --> <script> console.log('Hello from inline JS!'); </script> <!-- 3. defer vs async on external scripts --> <<!-- defer: runs after HTML parsed (keeps order) ✅ --> <script src="app.js" defer></script> <!-- async: runs as soon as downloaded (no order guarantee) --> <script src="analytics.js" async></script>

Selecting DOM Elements

// Modern: querySelector (returns first match) const btn = document.querySelector('.my-btn'); const hero = document.querySelector('#hero'); const input = document.querySelector('input[type="email"]'); // Returns all matches as NodeList const cards = document.querySelectorAll('.card'); // Iterate over NodeList cards.forEach(card => { card.style.background = '#eff6ff'; });

🖱️ Event Listeners

const btn = document.querySelector('#my-btn'); // addEventListener(event, callback) btn.addEventListener('click', function(event) { console.log('Button clicked!', event); }); // Arrow function shorthand btn.addEventListener('click', (e) => { e.preventDefault(); // stop default action (e.g. form submit) e.stopPropagation(); // don't bubble to parent }); // Common events // click, dblclick, mouseenter, mouseleave // keydown, keyup, keypress // submit, input, change, focus, blur // scroll, resize, load, DOMContentLoaded

🔧 Changing the DOM

const el = document.querySelector('#target'); // Read/write text content (safe — no HTML) el.textContent = 'Updated text!'; // Read/write HTML (use with trusted content only) el.innerHTML = '<strong>Bold text</strong>'; // Change styles directly el.style.color = '#1d4ed8'; el.style.display = 'none'; el.style.cssText = 'color: red; font-size: 1.2rem;'; // Read/write attributes el.setAttribute('aria-hidden', 'true'); el.getAttribute('href'); el.removeAttribute('disabled'); // Create and append new elements const newDiv = document.createElement('div'); newDiv.textContent = 'I was created by JS!'; document.body.appendChild(newDiv);

🎨 classList API — Toggling CSS Classes

const menu = document.querySelector('.menu'); // Add a class menu.classList.add('is-open'); // Remove a class menu.classList.remove('is-open'); // Toggle — adds if missing, removes if present menu.classList.toggle('is-open'); // Check if class exists if (menu.classList.contains('is-open')) { console.log('Menu is open!'); } /* In CSS: */ .menu { max-height: 0; overflow: hidden; transition: max-height .3s; } .menu.is-open { max-height: 400px; }
